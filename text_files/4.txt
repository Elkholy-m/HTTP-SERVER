bool test_case (int* vals, int len) {
    for (int i = 0; i < len; i++) {
        if (vals[i] < 1 || vals[i] > 6) {
            return false;
        }
    }

    return true;
}

void dump_arr(int* arr, int len) {
    for (int i = 0; i < len; i++) {
        printf("arr[%d] = %d\n", i, arr[i]);
    }
}

int main() {
    int arr[50];
    for (int i = 0; i < 50; i++) {
        srand(time(NULL));
        float r_num = rand() / (float)RAND_MAX;
        int range_num = r_num * 6 + 1;
        sleep(1);
        arr[i] = range_num;
    }

    if (test_case(arr, 50))
        printf("SUCCESS\n");
    else 
        printf("FAIL\n");

    dump_arr(arr, 50);

    return 0;
}
